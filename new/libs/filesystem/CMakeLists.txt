cmake_minimum_required(VERSION 2.8)
project(BoostFilesystem CXX)

find_package(Boost COMPONENTS cmake system NO_MODULE)
include(${Boost_USE_FILE})
include(BoostAddLibrary)
include(BoostExport)

boost_add_library(filesystem
  SOURCES
    src/operations.cpp
    src/path.cpp
    src/portability.cpp
    src/utf8_codecvt_facet.cpp
    src/codecvt_error_category.cpp
    src/path_traits.cpp
    src/unique_path.cpp
    src/windows_file_codecvt.cpp
  LINK_LIBRARIES
    ${Boost_LIBRARIES}
  )

boost_export(
  TARGETS
    filesystem
  BOOST_DEPENDS
    bind
    config
    core
    functional
    integer
    io
    iterator
    smart_ptr
    static_assert
    system
    #test
    #timer
    type_traits
    utility
  INCLUDE_DIRECTORIES
    include
  )
