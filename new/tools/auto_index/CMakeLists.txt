cmake_minimum_required(VERSION 2.8 FATAL_ERROR)
project(BoostAutoIndex CXX)

find_package(Boost
  COMPONENTS
    algorithm
    array
    core
    exception
    filesystem
    format
    program_options
    range
    regex
    smart_ptr
    utility
  NO_MODULE
  )
include(${Boost_USE_FILE})
include(BoostAddExecutable)

boost_add_executable(auto_index
  SOURCES
    src/auto_index.cpp
    src/file_scanning.cpp
    src/index_generator.cpp
    src/tiny_xml.cpp
  LINK_LIBRARIES
    ${Boost_LIBRARIES}
  )
